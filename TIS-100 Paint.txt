@0
#TIS-100 PAINT
#BY MEDIC5700

#'8462' TO MOVE
#CAN MOVE CURSER
#OUTSIDE OF
#CONSOLE BOUNDS

#'019' FOR COLOUR
# 0 BLACK
# 1 WHITE
# 9 RED

@1
#INITIALIZE COLOUR
MOV 1 RIGHT
MOV 1 DOWN
#INPUT LOOP
S:MOV UP ACC
MOV ACC DOWN
MOV ACC RIGHT
JMP S

@2
#COLOUR SELECTOR
S:MOV LEFT ACC
JNZ 1 #NOT BLACK
0:MOV 0 RIGHT #B
  JMP S
1:SUB 1
  JNZ 2 #NOT WHITE
    MOV 3 RIGHT #W
    JMP S
2:SUB 8
  JNZ E #NOT WRB
    MOV 4 RIGHT #R
    JMP S
#NOT COLOUR
E:MOV -1 RIGHT

@3
MOV LEFT DOWN

@4
MOV RIGHT DOWN

@5
S:MOV UP ACC
MOV ACC LEFT

SUB 4
JNZ R #NOT LEFT
  #MOVE LEFT
L:MOV -1 RIGHT
  JMP S
R:SUB 2
  JNZ E #NOT RIGHT
    #MOVE RIGHT
    MOV 1 RIGHT
    JMP S
  #NO X CHANGE
E:MOV 0 RIGHT

@6
#ACC KEEPS TRACK
#OF X POSITION
ADD LEFT
MOV ACC DOWN

@7
MOV UP DOWN

@8
S:MOV UP ACC
SUB 2
JNZ D #NOT UP
  #MOVE UP
U:MOV 1 RIGHT
  JMP S
D:SUB 6
  JNZ E #NOT DOWN
    #MOVE DOWN
    MOV -1 RIGHT
    JMP S
  #NO Y CHANGE
E:MOV 0 RIGHT

@9
#ACC KEEPS TRACK
#OF Y POSITION
ADD LEFT
MOV ACC RIGHT

@10
#X COORDINATE
MOV UP DOWN
#Y COORDINATE
MOV LEFT DOWN
#COLOUR VALUE
MOV RIGHT DOWN
#TERMINATING VALUE
MOV -1 DOWN

@11
#IF NOT COLOUR:
#SENDS LAST VALID
#COLOUR
MOV UP ACC
JLZ 1
  SAV
1:SWP
  SAV
  MOV ACC LEFT
